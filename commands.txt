git config --global user.name "user name"     [setting git user name]
git config --global user.email "user email"   [setting git user email]
git init                                      [to initialize in cwd(current working directosry)]
git status                                    [list all untracked files]
git add -A                                    [add all untracked files to staging area]
git commit -m "commit msg"                    [to commit with descriptive commit msg]
git remote add remote_name path               [adding path]
  eg:git remote add orgin https://github.com/Sandra-pm1/GitLearning.git
git push remote_name branch 
  eg:git push orgin master
git branch                                    [list branches]
